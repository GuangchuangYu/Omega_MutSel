############################# NYP dAIC tables ###################################
make_daic_table <- function(dat, fspecs, rownames)
{
for (i in seq(1,10,2)){
    mean_data <- round( mean(dat$aic[dat$fspec=="fnuc_data"] - dat$aic[dat$fspec==fspecs[i]]), 2)
    mean_true <- round( mean(dat$aic[dat$fspec=="fnuc_true"] - dat$aic[dat$fspec==fspecs[i+1]]), 2)
    sd_data   <- round( sd(dat$aic[dat$fspec=="fnuc_data"] - dat$aic[dat$fspec==fspecs[i]]), 2)
    sd_true   <- round( sd(dat$aic[dat$fspec=="fnuc_true"] - dat$aic[dat$fspec==fspecs[i+1]]), 2)
    data <- paste(mean_data, " (", sd_data, ") ", sep="")
    true <- paste(mean_true, " (", sd_true, ") ", sep="")
    cat(paste(rownames[(i+1)/2], dat$num_params[dat$fspec==fspecs[i]][1], data, true, sep=' & '))
    cat(' \\\\ \n')
}}

cat("%%%%%%%%%  NP dAIC table  %%%%%%%%%%\n")
make_daic_table(np, fspecs, rownames)
cat("\n\n\n%%%%%%%%%  Yeast dAIC table  %%%%%%%%%\n")
make_daic_table(yeast, fspecs, rownames)
cat("\n\n\n%%%%%%%%%  Polio dAIC table  %%%%%%%%%\n")
make_daic_table(polio, fspecs, rownames)
